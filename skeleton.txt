set -e
size=256
dd if=/dev/zero bs=1M count=$size of=${size}MB.disk
loop=$(losetup -f)
losetup -P $loop ${size}MB.disk
echo -e "n\n \n \n \n \n t\n b\n a\n w" | fdisk $loop
sleep 1; partprobe $loop
mkfs.vfat ${loop}p1
mkdir -p tmp
mount ${loop}p1 tmp
tar xvzf tcl-boot-syslinux.tgz -moC tmp
umount tmp
syslinux -d /boot/syslinux -i ${loop}p1
zcat tcl-usb-boot-enable.gz >$loop
losetup -d $loop
gzip -9c ${size}MB.disk >tcl-skeleton.disk.gz
rm -f ${size}MB.disk
